58  	 * OrderPerform, answers to: where does the player go?
59  	 */
60  	public void orderPerform(final ControllerOrder controllerOrder) {
61  		Entity player = this.model.getPlayer();
62  		int playerX = (int) player.getPosition().getX();
63  		int playerY = (int) player.getPosition().getY();
64  		switch (controllerOrder) {
65  			case UP:
66  				moveEntity(player, playerX, playerY - 1, controllerOrder);
67  				break;
68  			case DOWN:
69  				moveEntity(player, playerX, playerY + 1, controllerOrder);
70  				break;
71  			case RIGHT:
72  				moveEntity(player, playerX + 1, playerY, controllerOrder);
73  				break;
74  			case LEFT:
75  				moveEntity(player, playerX - 1, playerY, controllerOrder);
76  				break;
77  			default:
78  				break;
79  		}
80  	}
